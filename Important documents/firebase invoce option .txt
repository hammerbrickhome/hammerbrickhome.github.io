import React, { useState, useEffect, useMemo, useCallback } from 'react';
import { initializeApp } from 'firebase/app';
import { getAuth, signInAnonymously, signInWithCustomToken, onAuthStateChanged } from 'firebase/auth';
import { getFirestore, doc, setDoc, onSnapshot, collection, query, getDoc } from 'firebase/firestore';

// Global variables provided by the environment
const appId = typeof __app_id !== 'undefined' ? __app_id : 'default-app-id';
const firebaseConfig = typeof __firebase_config !== 'undefined' ? JSON.parse(__firebase_config) : null;
const initialAuthToken = typeof __initial_auth_token !== 'undefined' ? __initial_auth_token : null;

// Helper to format currency
const formatCurrency = (amount) => new Intl.NumberFormat('en-US', {
  style: 'currency',
  currency: 'USD',
  minimumFractionDigits: 2,
  maximumFractionDigits: 2,
}).format(amount);

// Utility to generate a secure, unguessable token for the client link
const generateClientToken = () => crypto.randomUUID() + crypto.randomUUID();

// Initial invoice data structure for a NEW invoice (id will be set on save)
const getNewInvoiceData = () => ({
  id: null, // Will be set by Firestore doc reference
  clientToken: generateClientToken(),
  status: 'Draft', // New status field
  invoiceNumber: 'INV-' + Math.floor(Math.random() * 10000),
  from: { name: 'Hammer Brick & Home LLC', address: '123 Masonry Way, Denver, CO' },
  to: { name: '', address: '' },
  date: new Date().toISOString().split('T')[0],
  dueDate: new Date(Date.now() + 30 * 24 * 60 * 60 * 1000).toISOString().split('T')[0], // 30 days later
  taxRate: 10, // Default 10%
  items: [
    { id: crypto.randomUUID(), description: 'Initial Estimate/Service', quantity: 1, price: 0.00 },
  ],
  notes: 'Payment terms: Due upon receipt. Please refer to our agreement.',
});

// Utility to create a new empty line item
const createNewItem = () => ({
  id: crypto.randomUUID(),
  description: '',
  quantity: 0,
  price: 0.00,
});


const InvoiceEditor = ({ invoice, setInvoice, totals, handleGeneralChange, handleItemChange, handleAddItem, handleDeleteItem, saveStatus, saveInvoice, InputField, TextAreaField }) => {

  const ShareLinkGenerator = () => {
    const handleCopy = () => {
      // In a real application, you would construct the portal URL here:
      // const clientPortalUrl = `https://yourdomain.com/portal/invoice/${invoice.id}?token=${invoice.clientToken}`;
      const mockUrl = `[Simulated Client Portal Link - Use ID: ${invoice.id}, Token: ${invoice.clientToken}]`;

      // Copy to clipboard (using document.execCommand for iFrame compatibility)
      const el = document.createElement('textarea');
      el.value = mockUrl;
      document.body.appendChild(el);
      el.select();
      document.execCommand('copy');
      document.body.removeChild(el);

      // Using alert as a temporary replacement for a UI message box
      // In a production app, replace this with a beautiful modal or toast notification.
      alert('Secure link copied to clipboard!\n\nSimulated Link: ' + mockUrl);
    };

    if (!invoice.id) return null; // Cannot share an unsaved invoice

    return (
      <div className="flex flex-col items-start gap-2 p-4 bg-yellow-50 border border-yellow-200 rounded-xl mt-6">
        <p className="font-bold text-yellow-700">Share Secure Link (Client Portal Ready)</p>
        <p className="text-sm text-yellow-600">This token secures the client's access to this specific invoice.</p>
        <div className="flex w-full">
            <span className="truncate text-xs font-mono bg-yellow-100 p-2 rounded-l-lg border-y border-l border-yellow-300 flex-grow">
              {invoice.clientToken}
            </span>
            <button
                onClick={handleCopy}
                className="px-4 py-2 text-sm bg-yellow-500 text-white font-semibold rounded-r-lg shadow-md hover:bg-yellow-600 transition duration-150"
            >
                Copy Link
            </button>
        </div>
      </div>
    );
  };

  return (
    <div className="p-6 md:p-10">
      <div className="flex justify-between items-start mb-8 border-b pb-4">
        <h1 className="text-3xl font-bold text-blue-700">Invoice Editor</h1>
        <div className="text-right text-xs text-gray-500">
          <p className="text-sm font-semibold text-gray-700">ID: {invoice.id || 'NEW DRAFT'}</p>
          <p className={`font-semibold transition-opacity duration-500 ${saveStatus ? 'opacity-100' : 'opacity-0'}`}>
            {saveStatus}
          </p>
        </div>
      </div>

      {/* Invoice Info & Dates */}
      <div className="grid grid-cols-2 md:grid-cols-4 gap-4 mb-8">
        <InputField label="Invoice Number" name="invoiceNumber" value={invoice.invoiceNumber} onChange={handleGeneralChange} />
        <InputField label="Issue Date" name="date" value={invoice.date} type="date" onChange={handleGeneralChange} />
        <InputField label="Due Date" name="dueDate" value={invoice.dueDate} type="date" onChange={handleGeneralChange} />
        <InputField label="Tax Rate (%)" name="taxRate" value={invoice.taxRate} type="number" onChange={handleGeneralChange} />
      </div>

      {/* Sender and Client Details */}
      <div className="grid grid-cols-1 md:grid-cols-2 gap-8 mb-10">
        <div className="p-4 border border-gray-200 rounded-xl bg-blue-50/50">
          <h2 className="text-lg font-semibold text-blue-600 mb-3">From (Your Details)</h2>
          <InputField label="Company Name" name="from.name" value={invoice.from.name} onChange={handleGeneralChange} />
          <TextAreaField label="Address" name="from.address" value={invoice.from.address} onChange={handleGeneralChange} />
        </div>

        <div className="p-4 border border-gray-200 rounded-xl bg-green-50/50">
          <h2 className="text-lg font-semibold text-green-600 mb-3">Bill To (Client Details)</h2>
          <InputField label="Client Name" name="to.name" value={invoice.to.name} onChange={handleGeneralChange} placeholder="Client Name (for unique ID)" />
          <TextAreaField label="Address" name="to.address" value={invoice.to.address} onChange={handleGeneralChange} />
        </div>
      </div>

      {/* Line Items Table */}
      <div className="overflow-x-auto shadow-md rounded-xl mb-8">
        <table className="min-w-full bg-white">
          <thead className="bg-gray-50 border-b">
            <tr>
              <th className="py-3 px-3 text-left text-xs font-medium text-gray-500 uppercase tracking-wider w-1/2">Description</th>
              <th className="py-3 px-3 text-left text-xs font-medium text-gray-500 uppercase tracking-wider w-1/6">Qty</th>
              <th className="py-3 px-3 text-left text-xs font-medium text-gray-500 uppercase tracking-wider w-1/6">Price</th>
              <th className="py-3 px-3 text-left text-xs font-medium text-gray-500 uppercase tracking-wider w-1/6">Total</th>
              <th className="py-3 px-3"></th>
            </tr>
          </thead>
          <tbody>
            {invoice.items.map(item => (
              <tr key={item.id} className="border-b hover:bg-gray-50">
                <td className="p-3">
                  <input
                    type="text"
                    value={item.description}
                    onChange={(e) => handleItemChange(item.id, 'description', e.target.value)}
                    className="w-full p-1 border rounded focus:ring-blue-500 focus:border-blue-500"
                    placeholder="Description"
                  />
                </td>
                <td className="p-3">
                  <input
                    type="number"
                    min="0"
                    value={item.quantity}
                    onChange={(e) => handleItemChange(item.id, 'quantity', e.target.value)}
                    className="w-full p-1 border rounded text-right focus:ring-blue-500 focus:border-blue-500"
                  />
                </td>
                <td className="p-3">
                  <input
                    type="number"
                    min="0"
                    step="0.01"
                    value={item.price.toFixed(2)}
                    onChange={(e) => handleItemChange(item.id, 'price', e.target.value)}
                    className="w-full p-1 border rounded text-right focus:ring-blue-500 focus:border-blue-500"
                  />
                </td>
                <td className="p-3 font-medium text-gray-700 text-right">
                  {formatCurrency(item.quantity * item.price)}
                </td>
                <td className="p-3 text-center">
                  <button
                    onClick={() => handleDeleteItem(item.id)}
                    className="text-red-500 hover:text-red-700 transition duration-150"
                    title="Delete Item"
                  >
                    <svg xmlns="http://www.w3.org/2000/svg" className="h-5 w-5" viewBox="0 0 20 20" fill="currentColor">
                      <path fillRule="evenodd" d="M9 2a1 1 0 00-.894.553L7.382 4H4a1 1 0 000 2v10a2 2 0 002 2h8a2 2 0 002-2V6a1 1 0 100-2h-3.382l-.724-1.447A1 1 0 0011 2H9zM7 8a1 1 0 012 0v6a1 1 0 11-2 0V8zm4 0a1 1 0 10-2 0v6a1 1 0 102 0V8z" clipRule="evenodd" />
                    </svg>
                  </button>
                </td>
              </tr>
            ))}
          </tbody>
        </table>
      </div>

      <button
        onClick={handleAddItem}
        className="px-4 py-2 bg-blue-500 text-white font-semibold rounded-lg shadow-md hover:bg-blue-600 transition duration-150 flex items-center mb-10"
      >
        <svg xmlns="http://www.w3.org/2000/svg" className="h-5 w-5 mr-2" viewBox="0 0 20 20" fill="currentColor">
          <path fillRule="evenodd" d="M10 5a1 1 0 011 1v3h3a1 1 0 110 2h-3v3a1 1 0 11-2 0v-3H6a1 1 0 110-2h3V6a1 1 0 011-1z" clipRule="evenodd" />
        </svg>
        Add Line Item
      </button>

      {/* Share Link */}
      <ShareLinkGenerator />

      {/* Totals and Notes */}
      <div className="grid grid-cols-1 md:grid-cols-3 gap-8 mt-10">
        <div className="md:col-span-2">
          <TextAreaField label="Notes / Payment Terms" name="notes" value={invoice.notes} onChange={handleGeneralChange} />
        </div>

        <div className="space-y-3 p-4 bg-gray-50 rounded-xl border border-gray-200 shadow-inner">
          <div className="flex justify-between font-medium text-gray-600">
            <span>Subtotal:</span>
            <span>{formatCurrency(totals.subtotal)}</span>
          </div>

          <div className="flex justify-between font-medium text-gray-600">
            <span>Tax ({invoice.taxRate}%):</span>
            <span>{formatCurrency(totals.taxAmount)}</span>
          </div>

          <div className="flex justify-between pt-3 border-t-2 border-blue-200">
            <span className="text-xl font-bold text-blue-700">GRAND TOTAL:</span>
            <span className="text-xl font-bold text-blue-700">{formatCurrency(totals.total)}</span>
          </div>
        </div>
      </div>
    </div>
  );
};

  <div className="p-6 md:p-10">
    <div className="flex justify-between items-center mb-8 border-b pb-4">
      <h1 className="text-3xl font-bold text-gray-800">Invoice Dashboard ({invoices.length} Total)</h1>

            {saveStatus}
          </p>
        </div>

    >
      <svg xmlns="http://www.w3.org/2000/svg" className="h-5 w-5 mr-2" viewBox="0 0 20 20" fill="currentColor">

const App = () => {
  const [invoices, setInvoices] = useState([]); // List of all invoices (metadata)
  const [invoice, setInvoice] = useState(getNewInvoiceData); // Currently edited invoice data
  const [db, setDb] = useState(null);
  const [userId, setUserId] = useState(null);
  const [isAuthReady, setIsAuthReady] = useState(false);
  const [saveStatus, setSaveStatus] = useState('');
  const [currentView, setCurrentView] = useState('dashboard'); // 'dashboard' or 'editor'

  // --- FIREBASE INITIALIZATION AND AUTHENTICATION ---
  useEffect(() => {
    if (!firebaseConfig) return;

    try {
      const app = initializeApp(firebaseConfig);
      const firestore = getFirestore(app);
      const authService = getAuth(app);
      setDb(firestore);

      const authenticate = async () => {
        try {
          if (initialAuthToken) {
            await signInWithCustomToken(authService, initialAuthToken);
          } else {
            await signInAnonymously(authService);
          }
        } catch (error) {
          console.error("Firebase Auth failed:", error);
          await signInAnonymously(authService);
        }
      };

      const unsubscribe = onAuthStateChanged(authService, (user) => {
        if (user) {
          setUserId(user.uid);
          setIsAuthReady(true);
        } else {
          if (!userId) authenticate();
        }
      });

      authenticate();
      return () => unsubscribe();
    } catch (error) {
      console.error("Failed to initialize Firebase:", error);
    }
  }, [firebaseConfig, initialAuthToken, userId]);


  // --- FIRESTORE DATA LISTENER (FETCHING ALL INVOICES FOR DASHBOARD) ---
  useEffect(() => {
    if (!db || !userId || !isAuthReady) return;

    const invoicesCollectionRef = collection(db, 'artifacts', appId, 'users', userId, 'invoices');
    const q = query(invoicesCollectionRef);

    // Listener for the dashboard list
    const unsubscribe = onSnapshot(q, (snapshot) => {
      const invoicesList = [];
      snapshot.forEach(docSnap => {
        const data = docSnap.data();
        const subtotal = data.items?.reduce((sum, item) => sum + ((item.quantity || 0) * (item.price || 0)), 0) || 0;
        const taxAmount = subtotal * ((data.taxRate || 0) / 100);
        const total = subtotal + taxAmount;
        
        invoicesList.push({
          id: docSnap.id,
          invoiceNumber: data.invoiceNumber,
          to: data.to,
          date: data.date,
          dueDate: data.dueDate,
          status: data.status,
          total: total, // Calculated for display
        });
      });
      setInvoices(invoicesList);
      console.log(`Loaded ${invoicesList.length} invoices for the dashboard.`);

      // If we are currently in the editor, ensure the data is synced
      if (currentView === 'editor' && invoice.id) {
          const latestData = invoicesList.find(i => i.id === invoice.id);
          if (latestData) {
            // No need to call loadInvoice here, as this useEffect runs whenever currentView changes
            // and we rely on the debounced save for live updates within the editor.
          }
      }

    }, (error) => {
      console.error("Error fetching invoices list:", error);
    });

    return () => unsubscribe();
  }, [db, userId, isAuthReady, currentView]); // currentView dependency ensures re-fetch if we switch back


  // --- CALCULATIONS ---
  const totals = useMemo(() => {
    const subtotal = invoice.items.reduce((sum, item) => sum + ((item.quantity || 0) * (item.price || 0)), 0);
    const taxAmount = subtotal * ((invoice.taxRate || 0) / 100);
    const total = subtotal + taxAmount;
    return { subtotal, taxAmount, total };
  }, [invoice.items, invoice.taxRate]);


  // --- HANDLERS ---

  // Update a single field in the main invoice object (e.g., from, to, date, taxRate)
  const handleGeneralChange = useCallback((e) => {
    const { name, value } = e.target;
    if (name.includes('.')) {
      const [parent, child] = name.split('.');
      setInvoice(prev => ({
        ...prev,
        [parent]: { ...prev[parent], [child]: value }
      }));
    } else {
      setInvoice(prev => ({ ...prev, [name]: value }));
    }
  }, []);

  // Update a field for a specific line item
  const handleItemChange = useCallback((id, field, value) => {
    setInvoice(prev => ({
      ...prev,
      items: prev.items.map(item =>
        item.id === id ? { ...item, [field]: field === 'quantity' || field === 'price' ? parseFloat(value) || 0 : value } : item
      ),
    }));
  }, []);

  // Add a new line item
  const handleAddItem = useCallback(() => {
    setInvoice(prev => ({
      ...prev,
      items: [...prev.items, createNewItem()],
    }));
  }, []);

  // Delete a line item
  const handleDeleteItem = useCallback((id) => {
    setInvoice(prev => ({
      ...prev,
      items: prev.items.filter(item => item.id !== id),
    }));
  }, []);

  // --- NAVIGATION HANDLERS ---
  const createNewInvoice = useCallback(() => {
    setInvoice(getNewInvoiceData()); // Start with a fresh, unsaved invoice
    setCurrentView('editor');
  }, []);

  const loadInvoice = useCallback(async (invoiceId) => {
    if (!db || !userId) return;

    try {
      const docRef = doc(db, 'artifacts', appId, 'users', userId, 'invoices', invoiceId);
      const docSnap = await getDoc(docRef);

      if (docSnap.exists()) {
        const data = docSnap.data();
        setInvoice({ ...data, id: docSnap.id });
        setCurrentView('editor');
        console.log(`Loaded invoice ${invoiceId}.`);
      } else {
        console.error("Invoice not found:", invoiceId);
      }
    } catch (error) {
      console.error("Error loading invoice:", error);
    }
  }, [db, userId]);


  // --- SAVE FUNCTIONALITY ---
  const saveInvoice = useCallback(async (currentInvoice) => {
    if (!db || !userId) {
      setSaveStatus('Error: Not authenticated. Cannot save.');
      return;
    }

    // Prepare data for Firestore
    const invoiceData = {
      ...currentInvoice,
      items: currentInvoice.items.map(({ id, description, quantity, price }) => ({
        id,
        description,
        quantity: quantity || 0,
        price: price || 0,
      })),
      taxRate: parseFloat(currentInvoice.taxRate) || 0,
      total: totals.total, // Save the calculated total for dashboard use
      status: currentInvoice.status || 'Draft',
    };

    let docRef;
    if (currentInvoice.id) {
        // Update existing document
        docRef = doc(db, 'artifacts', appId, 'users', userId, 'invoices', currentInvoice.id);
    } else {
        // Create new document, let Firestore generate the unique ID within the collection
        docRef = doc(collection(db, 'artifacts', appId, 'users', userId, 'invoices'));
    }

    try {
      setSaveStatus('Saving...');
      await setDoc(docRef, invoiceData, { merge: true });
      
      // If a new document was created, update the local state with the new ID
      if (!currentInvoice.id) {
        setInvoice(prev => ({ ...prev, id: docRef.id }));
      }
      setSaveStatus('Saved!');
    } catch (error) {
      console.error("Error saving invoice:", error);
      setSaveStatus('Save Failed!');
    }
    setTimeout(() => setSaveStatus(''), 3000);
  }, [db, userId, totals.total]);

  // Effect to automatically save invoice data whenever it changes
  useEffect(() => {
    // Debounce to prevent excessive writes
    const handler = setTimeout(() => {
      if (currentView === 'editor' && isAuthReady && db && userId) {
        saveInvoice(invoice);
      }
    }, 1000); // 1 second debounce

    return () => {
      clearTimeout(handler);
    };
  }, [invoice, isAuthReady, db, userId, saveInvoice, currentView]);


  // --- UI Components ---

  const InputField = ({ label, name, value, type = 'text', onChange, placeholder = '' }) => (
    <div className="flex flex-col mb-4">
      <label className="text-xs font-semibold text-gray-500 mb-1">{label}</label>
      <input
        type={type}
        name={name}
        value={value}
        onChange={onChange}
        placeholder={placeholder}
        className="p-2 border border-gray-300 rounded-lg focus:ring-blue-500 focus:border-blue-500 transition duration-150 shadow-sm"
      />
    </div>
  );

  const TextAreaField = ({ label, name, value, onChange, placeholder = '' }) => (
    <div className="flex flex-col mb-4">
      <label className="text-xs font-semibold text-gray-500 mb-1">{label}</label>
      <textarea
        name={name}
        value={value}
        onChange={onChange}
        placeholder={placeholder}
        rows="3"
        className="p-2 border border-gray-300 rounded-lg focus:ring-blue-500 focus:border-blue-500 transition duration-150 shadow-sm resize-none"
      />
    </div>
  );


  // Simple Loading/Auth State
  if (!isAuthReady) {
    return (
      <div className="flex items-center justify-center min-h-screen bg-gray-50">
        <div className="text-center p-6 bg-white rounded-xl shadow-2xl">
          <svg className="animate-spin -ml-1 mr-3 h-5 w-5 text-blue-500 inline-block" xmlns="http://www.w3.org/2000/svg" fill="none" viewBox="0 0 24 24">
            <circle className="opacity-25" cx="12" cy="12" r="10" stroke="currentColor" strokeWidth="4"></circle>
            <path className="opacity-75" fill="currentColor" d="M4 12a8 8 0 018-8V0C5.373 0 0 5.373 0 12h4zm2 5.291A7.962 7.962 0 014 12H0c0 3.042 1.135 5.824 3 7.938l3-2.647z"></path>
          </svg>
          <p className="mt-2 text-gray-700 font-medium">Connecting to secure data storage...</p>
        </div>
      </div>
    );
  }

  return (
    <div className="min-h-screen bg-gray-100 p-4 sm:p-8 font-['Inter']">
      <div className="max-w-4xl mx-auto bg-white rounded-2xl shadow-2xl">
        
        {/* Navigation Bar */}
        <div className="flex border-b border-gray-200 bg-gray-50 rounded-t-2xl">
          <button
            onClick={() => setCurrentView('dashboard')}
            className={`py-3 px-6 font-semibold transition-colors duration-200 ${
              currentView === 'dashboard' ? 'bg-white text-blue-700 border-b-2 border-blue-700' : 'text-gray-500 hover:bg-gray-100'
            }`}
          >
            Dashboard
          </button>
          {currentView === 'editor' && (
            <button
              className={`py-3 px-6 font-semibold transition-colors duration-200 bg-white text-blue-700 border-b-2 border-blue-700`}
            >
              Invoice Editor
            </button>
          )}
        </div>

        {/* Main View Area */}
        {currentView === 'dashboard' ? (
          <InvoiceDashboard
            invoices={invoices}
            userId={userId}
            loadInvoice={loadInvoice}
            createNewInvoice={createNewInvoice}
            saveStatus={saveStatus}
            InputField={InputField}
            TextAreaField={TextAreaField}
          />
        ) : (
          <InvoiceEditor
            invoice={invoice}
            setInvoice={setInvoice}
            totals={totals}
            handleGeneralChange={handleGeneralChange}
            handleItemChange={handleItemChange}
            handleAddItem={handleAddItem}
            handleDeleteItem={handleDeleteItem}
            saveStatus={saveStatus}
            saveInvoice={saveInvoice}
            InputField={InputField}
            TextAreaField={TextAreaField}
          />
        )}
      </div>
    </div>
  );
};

export default App;
